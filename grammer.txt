INPUT -> (VAR_DECL | CONST_DECL)*
VAR_DECL -> IDENT = EXPR
CONST_DECL -> IDENT :: EXPR

FN_CALL -> IDENT(EXPR,*)
TYPE_DEF -> STRUCT | UNION | ENUM

EXPR -> A
A -> B (addminus B)*
B -> C (muldivmod C)*
C -> D (.D)*
D -> NUMBER | STRING | BOOLEAN | FN_CALL | IDENT | '(' expr ')' | TYPE_DEF | IDENT++ | IDENT--
